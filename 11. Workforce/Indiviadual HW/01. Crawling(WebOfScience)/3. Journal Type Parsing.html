def journal_type():
    chrome_options = Options()
    chrome_options.add_argument('--lang=en-us')
    driver = webdriver.Chrome(chrome_options=chrome_options,executable_path=r"C:/Users/seong/Downloads/chrome/chromedriver")
    
    #extract the unique publications urls with the unique journal types
    WOS_df = pd.read_pickle('./data/WOS_scraped_df_ver2.pkl')
    journal_type_list = []
    pub_urls_list = []
    for journal, publ_url in zip(WOS_df.journal_type, WOS_df.WOS_URL):
        if not journal in journal_type_list:
            pub_urls_list.append(publ_url) #set url 439
            journal_type_list.append(journal) #set journal type 439

    try:
        journal_type_df = pd.read_csv('./data/journal_type_version3.csv')
        start_idx = journal_type_list.index(list(journal_type_df.journal_type_main_page)[-1])+1 #find the index for a start
    except:
        journal_type_df = pd.DataFrame(columns = ['publication_url','journal_url','journal_type_main_page','journal_type_report_page', 'year', 'ttl_cites', 'journal_impact_factor', 'impact_factor_without_self_sites','5yr_impact_factor', 'immediacy_index', 'citable_item', 'cited_half_life', 'citing_half_life', 'eigenfactor_score', 'artical_infl_score', '%article_in_citable_items', 'normed_eigenfactor', 'avg_JIF_percentile'])
        start_idx = 0
    
    web_of_science_links = pub_urls_list[start_idx:]

    for w_link in web_of_science_links: 
        #Log-in on Web of Science
        print("New Journal Type: ", journal_type_list[start_idx])
        #Select the id and pass at random
        ids = ['']
        selected_id = random.choice(ids)
        passwords = ['']
        following_pass = passwords[ids.index(selected_id)]

        driver.get('https://login.webofknowledge.com/error/Error?PathInfo=%2F&Error=IPError')
        driver.implicitly_wait(10)
        driver.find_element_by_name('username').send_keys(selected_id) #email id
        driver.find_element_by_name('password').send_keys(following_pass) #password
        driver.implicitly_wait(10)
        driver.find_element_by_xpath('//*[@id="WoKerror"]/div/table[2]/tbody/tr/td[2]/form[1]/table/tbody/tr[3]/td[2]/a/button').click() #click
        try:
            driver.find_element_by_xpath('//*[@id="WoKerror"]/div/table[2]/tbody/tr/td[2]/p/a[1]').click() #exception log-in
        except:
            pass

        # Reaching the publication page on Web of Science
        driver.get(w_link)

        # Click the Journal Citation Reports button, which is connected to a journal information page
        try:
            driver.find_element_by_xpath('//*[@id="links_isi_product_1"]/li/a').click()
        except:
            start_idx = start_idx + 1
            continue

        driver.implicitly_wait(10)    
        
        
        first_tab = driver.window_handles[0]
        second_tab = driver.window_handles[-1]

        
        #Log in the Journal Citation Reports
        try:
            driver.switch_to.window(second_tab)
            driver.find_element_by_id('username').send_keys('')
            driver.find_element_by_id('password').send_keys('')
            driver.find_element_by_xpath('/html/body/div/div[1]/div[3]/div[1]/div/div/div/form[1]/button').click()
        except:
            pass

        driver.implicitly_wait(20)  

        try:
            driver.find_element_by_class_name('btn.btn-secondary').click()
        except:
            pass

        driver.implicitly_wait(20)
        
        # Select the main browser tab
        first_tab = driver.window_handles[0]
        new_tab = driver.window_handles[-1]
        driver.switch_to.window(new_tab)
        driver.implicitly_wait(10)

        web_soup2 = BeautifulSoup(driver.page_source, 'html.parser')
        source_val = web_soup2.select("tbody#gridview-1012-body > tr > td > div")
        
        #fill the parsed data in the dataframe
        row = journal_type_df.shape[0] #initialize the row index
        journal_type_df.loc[row, journal_type_df.columns[0]] = w_link
        journal_type_df.loc[row, journal_type_df.columns[1]] = driver.current_url
        journal_type_df.loc[row, journal_type_df.columns[2]] = journal_type_list[start_idx]
        try:
            journal_type_df.loc[row, journal_type_df.columns[3]] = web_soup2.select_one("h2#journalTitleData").string.strip()
        except:
            journal_type_df.loc[row, journal_type_df.columns[3]] = np.nan
        val_count = 0
        for val, col in zip(source_val, list(journal_type_df.columns[4:])*len(source_val)):
            if val_count == len(list(journal_type_df.columns[4:])):
                val_count = 0
                row = row + 1
                journal_type_df.loc[row, journal_type_df.columns[0]] = w_link
                journal_type_df.loc[row, journal_type_df.columns[1]] = driver.current_url
                journal_type_df.loc[row, journal_type_df.columns[2]] = journal_type_list[start_idx]
                try:
                    journal_type_df.loc[row, journal_type_df.columns[3]] = web_soup2.select_one("h2#journalTitleData").string.strip()
                except:
                    journal_type_df.loc[row, journal_type_df.columns[3]] = np.nan
            try:
                journal_type_df.loc[row, col] = int(val.text.replace(',', ''))
            except:
                if 'Not Available' in val.text.replace(',', ''):
                    journal_type_df.loc[row, col] = np.nan
                elif 'NA' in val.text.replace(',', ''):
                    journal_type_df.loc[row, col] = np.nan
                elif 'undefined' in val.text.replace(',', ''): 
                    journal_type_df.loc[row, col] = np.nan
                else:
                    journal_type_df.loc[row, col] = val.text

            val_count = val_count + 1

        journal_type_df.to_csv('./data/journal_type_version3.csv', index= None)
        start_idx = start_idx + 1
        driver.close()
        driver.switch_to.window(first_tab)
    driver.quit()
